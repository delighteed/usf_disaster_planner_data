Context: there is a web application "disaster planner" designed to help care givers and care recipients with special needs to evacuate from state of Florida in the event of hurricanes and flooding. There are a separate repositories for fronend and backend code.

Here are the 3 tables used for the application: caregiver, carerecipient, preparation. The below is the design of the tables: 

-- Table: public.caregiver

(
    caregiver_id integer NOT NULL DEFAULT nextval('caregiver_caregiver_id_seq'::regclass),
    first_name character varying(50) COLLATE pg_catalog."default" NOT NULL,
    last_name character varying(50) COLLATE pg_catalog."default" NOT NULL,
    age integer NOT NULL,
    gender character(1) COLLATE pg_catalog."default" NOT NULL,
    phone character(10) COLLATE pg_catalog."default",
    email character varying(50) COLLATE pg_catalog."default",
    street character varying(50) COLLATE pg_catalog."default",
    city character varying(50) COLLATE pg_catalog."default",
    state character(2) COLLATE pg_catalog."default",
    zip character varying(5) COLLATE pg_catalog."default",
    create_date timestamp without time zone,
    CONSTRAINT caregiver_pkey PRIMARY KEY (caregiver_id)
)


-- Table: public.carerecipient

(
    carerecipient_id integer NOT NULL DEFAULT nextval('carerecipient_carerecipient_id_seq'::regclass),
    first_name character varying(50) COLLATE pg_catalog."default" NOT NULL,
    last_name character varying(50) COLLATE pg_catalog."default" NOT NULL,
    age integer NOT NULL,
    gender character(1) COLLATE pg_catalog."default" NOT NULL,
    condition character varying(50) COLLATE pg_catalog."default" NOT NULL,
    special_needs character varying(500) COLLATE pg_catalog."default",
    create_date timestamp without time zone,
    caregiver_id integer,
    CONSTRAINT carerecipient_pkey PRIMARY KEY (carerecipient_id),
    CONSTRAINT carerecipient_caregiver_id_fkey FOREIGN KEY (caregiver_id)
        REFERENCES public.caregiver (caregiver_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

-- Table: public.preparation

(
    preparation_id integer NOT NULL DEFAULT nextval('preparation_preparation_id_seq'::regclass),
    mobile_home character(1) COLLATE pg_catalog."default" NOT NULL,
    flood_zone character(1) COLLATE pg_catalog."default",
    serious_flood_zone character(1) COLLATE pg_catalog."default" NOT NULL,
    area character(1) COLLATE pg_catalog."default" NOT NULL,
    garage character(1) COLLATE pg_catalog."default" NOT NULL,
    garage_door character(1) COLLATE pg_catalog."default",
    covers character(1) COLLATE pg_catalog."default" NOT NULL,
    evacuation_place character varying(50) COLLATE pg_catalog."default",
    other_place character varying(50) COLLATE pg_catalog."default",
    evacuation_vehicle character varying(50) COLLATE pg_catalog."default",
    clear_route character(1) COLLATE pg_catalog."default",
    time_to_destination integer,
    fuel integer,
    care_recipient_wait_time integer,
    purchased_flight_tickets character(1) COLLATE pg_catalog."default",
    purchased_bus_tickets character(1) COLLATE pg_catalog."default",
    favorite_foods character varying(50) COLLATE pg_catalog."default",
    fun_items character varying(50) COLLATE pg_catalog."default",
    favorite_items character varying(50) COLLATE pg_catalog."default",
    other_items character varying(50) COLLATE pg_catalog."default",
    create_date timestamp without time zone NOT NULL,
    caregiver_id integer,
    CONSTRAINT preparation_pkey PRIMARY KEY (preparation_id),
    CONSTRAINT preparation_caregiver_id_fkey FOREIGN KEY (caregiver_id)
        REFERENCES public.caregiver (caregiver_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)
